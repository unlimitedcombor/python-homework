题目 3.1: 判断闰年


#3.1
year = int(input())
if (year % 400 == 0) or (year % 4 == 0 and year % 100 != 0):
    print("闰年")
else:
    print("非闰年")

题目 3.2: 计算支付金额


#3.2
x = float(input())
if x <= 200:
    pay = x
elif x <= 500:
    pay = x * 0.9
else:
    pay = x * 0.8

# 去除可能的“.0”尾部
if pay.is_integer():
    print(int(pay))
else:
    print(pay)
