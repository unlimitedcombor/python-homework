题目 32.1: 求三个数中的最大值


#32.1
a, b, c = map(int, input().split())
max_val = a
if b > max_val:
    max_val = b
if c > max_val:
    max_val = c
print(max_val)

题目 32.4: 整数的位数及各位数字处理


#32.4
num_str = input().strip()
# a) 位数
print(len(num_str))
# b) 顺序输出各位数字
print(" ".join(num_str))
# c) 逆序输出各位数字
print(" ".join(num_str[::-1]))

题目 32.5: 解一元二次方程


#32.5
import math, cmath

a, b, c = map(float, input().split())

if a == 0:
    print("非二次方程")
else:
    d = b*b - 4*a*c
    if d > 0:
        rd = math.sqrt(d)
        r1 = (-b + rd) / (2*a)
        r2 = (-b - rd) / (2*a)
        print(f"{r1} {r2}")
    elif d == 0:
        r = -b / (2*a)
        print(f"{r} {r}")
    else:
        rd = cmath.sqrt(d)
        r1 = (-b + rd) / (2*a)
        r2 = (-b - rd) / (2*a)
        print(f"{r1} {r2}")
