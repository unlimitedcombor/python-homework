题目 1: 统计随机整数出现次数


#1
import random

# 设置随机种子
random.seed(1)

# 生成1000个随机整数，范围在[20, 100]之间
numbers = [random.randint(20, 100) for _ in range(1000)]

# 统计每个数字出现的次数
count_dict = {}
for num in numbers:
    if num in count_dict:
        count_dict[num] += 1
    else:
        count_dict[num] = 1

# 按升序排列并输出
for num in sorted(count_dict.keys()):
    print(f"{num}:{count_dict[num]}次", end=" ")

题目 2: 模拟购物车功能


#2
products = [['华为P30', 6688], ['wine', 1488], ['Book', 80], ['Shoes', 799], ['MateBook', 5800]]

# 打印商品列表
for i, product in enumerate(products, start=1):
    print(f"{i}.{product[0]}:{product[1]}")

cart = []
while True:
    input_num = input()
    try:
        num = int(input_num)
        if 1 <= num <= 5:
            if num - 1 in cart:
                print("已添加到购物车")
            else:
                cart.append(num - 1)
                print("添加成功")
        else:
            print("商品不存在")
    except ValueError:
        break

# 输出购物车信息
for index in cart:
    print(f"{products[index][0]}:{products[index][1]}")

题目 3: 字母转换为电话按键数字

# 定义字母到数字的映射关系
letter_to_number = {
    'a': '2', 'b': '2', 'c': '2',
    'd': '3', 'e': '3', 'f': '3',
    'g': '4', 'h': '4', 'i': '4',
    'j': '5', 'k': '5', 'l': '5',
    'm': '6', 'n': '6', 'o': '6',
    'p': '7', 'q': '7', 'r': '7', 's': '7',
    't': '8', 'u': '8', 'v': '8',
    'w': '9', 'x': '9', 'y': '9', 'z': '9'
}

# 输入英文字符串
input_str = input()

# 转换为数字字符串
result = ''.join([letter_to_number[char] for char in input_str if char.isalpha()])

# 输出结果
print(result)

题目 4: 统计《水浒传》人物出现次数


#4
# 指定的名字列表
names = ['宋江', '卢俊义', '吴用', '公孙胜', '关胜', '林冲', '秦明', '呼延灼', '花荣', '柴进', 
         '李应', '朱仝', '鲁智深', '武松', '董平', '张清', '杨志', '徐宁', '索超', '戴宗']

# 读取文件内容
wj = 'shuihuzhuan.txt'  # 文件名
txt = open(wj, "r", encoding='utf-8').read()

# 初始化一个字典来存储人物名称的出现次数
name_count = {name: 0 for name in names}

# 统计每个人物的出现次数
for name in names:
    name_count[name] = txt.count(name)

# 按照出现次数进行升序排序
sorted_names = sorted(name_count.items(), key=lambda x: x[1])

# 输出结果
for name, count in sorted_names:
    print(f"{name} 出现次数：{count}")
