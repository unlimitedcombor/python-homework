题目 1: 计算能喝到的汽水总数


#1
def calculate_bottles(money):
    total = money  # 初始用钱买的水
    empty_bottles = total  # 初始空瓶数
    caps = total     # 初始瓶盖数
    
    # 只要有空瓶或瓶盖可以兑换就继续
    while empty_bottles >= 2 or caps >= 3:
        # 用空瓶兑换水
        exchanged_from_bottles = empty_bottles // 2
        total += exchanged_from_bottles
        empty_bottles = empty_bottles % 2 + exchanged_from_bottles
        caps += exchanged_from_bottles
        
        # 用瓶盖兑换水
        exchanged_from_caps = caps // 3
        total += exchanged_from_caps
        caps = caps % 3 + exchanged_from_caps
        empty_bottles += exchanged_from_caps
    
    return total

# 输入金额
money = int(input())
print(calculate_bottles(money))

题目 2: 计算三个整数的最大公约数


#2
import math

def gcd_three_numbers(a, b, c):
    try:
        # 计算a, b和c的最大公约数
        return math.gcd(math.gcd(a, b), c)
    except Exception:
        # 如果发生任何异常，返回“发生异常”
        return "发生异常"

# 输入三个整数
try:
    num1 = int(input())
    num2 = int(input())
    num3 = int(input())
    
    # 计算并输出最大公约数
    print(gcd_three_numbers(num1, num2, num3))
except ValueError:
    # 如果输入的不是有效整数，输出异常信息
    print("发生异常")

题目 3: 计算阶乘和


#3
import math

def factorial_sum(n):
    # 计算1! + 2! + ... + n!
    total = 0
    for i in range(1, n+1):
        total += math.factorial(i)
    return total

# 输入最大的数字n
try:
    n = int(input())
    
    # 输出阶乘和
    print(factorial_sum(n))
except ValueError:
    print("发生异常")

题目 4: 输出指定范围内的闰年


#4
def is_leap_year(year):
    if (year % 4 == 0 and year % 100 != 0) or (year % 400 == 0):
        return True
    return False

def print_leap_years(start, end):
    leap_years = []
    for year in range(start, end + 1):
        if is_leap_year(year):
            leap_years.append(str(year))
    print(" ".join(leap_years))

print_leap_years(1980, 2050)

题目 5: 计算某月的天数


#5
import calendar

def days_in_month(y, m):
    return calendar.monthrange(y, m)[1]

# 输入年份和月份
y = int(input())
m = int(input())

# 输出该月的天数
print(days_in_month(y, m))

题目 6: 将一行单词首字母大写


#6
def capitalize_words(word_list_str):
    try:
        return " ".join([word.capitalize() for word in word_list_str.split()])
    except Exception:
        return "发生异常"

# 输入一行单词
input_str = input()

# 输出转换后的单词
print(capitalize_words(input_str))
